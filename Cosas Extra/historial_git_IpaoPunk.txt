+		$this->tenant_id     = IPAO_Settings::get( 'tenant_id' );
+		$this->client_id     = IPAO_Settings::get( 'client_id' );
+		$this->client_secret = IPAO_Settings::get( 'client_secret' );

+	public function is_configured() {
+		return $this->tenant_id && $this->client_id && $this->client_secret
+			&& $this->drive_id && $this->folder_id;
+	}
+
+	private function get_access_token() {
+		if ( $this->access_token ) {
+			return $this->access_token;
+		}
+
+		$url  = "https://login.microsoftonline.com/{$this->tenant_id}/oauth2/v2.0/token";
+		$body = [
+			'grant_type'    => 'client_credentials',
+			'client_id'     => $this->client_id,
+			'client_secret' => $this->client_secret,
+			'scope'         => 'https://graph.microsoft.com/.default',
+		];
+
+		$response = wp_remote_post( $url, [
+			'body'    => $body,
+			'timeout' => 30,
+		] );
+
+		if ( is_wp_error( $response ) ) {
+			throw new Exception( 'Error obteniendo token: ' . $response->get_error_message() );
+		}
+
+		$data = json_decode( wp_remote_retrieve_body( $response ), true );
+		if ( empty( $data['access_token'] ) ) {
+			throw new Exception( 'No se pudo obtener el token de acceso de Microsoft Graph.' );
+		}
+
+		$this->access_token = $data['access_token'];
+		return $this->access_token;
+	}
+
+	/**
+	 * Crea una subcarpeta dentro de la carpeta raíz configurada.
+	 * Devuelve el ID de la nueva carpeta.
+	 */
+	public function create_folder( $folder_name ) {
+		$token = $this->get_access_token();
+		$url   = "https://graph.microsoft.com/v1.0/drives/{$this->drive_id}/items/{$this->folder_id}/children";
+
+		$response = wp_remote_post( $url, [
+			'headers' => [
+				'Authorization' => 'Bearer ' . $token,
+				'Content-Type'  => 'application/json',
+			],
+			'body'    => json_encode( [
+				'name'                              => $folder_name,
+				'folder'                            => new stdClass(),
+				'@microsoft.graph.conflictBehavior' => 'rename',
+			] ),
+			'timeout' => 30,
+		] );
+
+		if ( is_wp_error( $response ) ) {
+			throw new Exception( 'Error creando carpeta en SharePoint: ' . $response->get_error_message() );
+		}
+
+		$data = json_decode( wp_remote_retrieve_body( $response ), true );
+		if ( empty( $data['id'] ) ) {
+			$err = isset( $data['error']['message'] ) ? $data['error']['message'] : wp_remote_retrieve_body( $response );
+			throw new Exception( 'No se pudo crear la carpeta en SharePoint: ' . $err );
+		}
+
+		return $data['id'];
+	}

+	/**
+	 * Sube un archivo a una carpeta (por ID de carpeta).

+		$token        = $this->get_access_token();

+
+		$response = wp_remote_request( $url, [
+			'method'  => 'PUT',
+			'headers' => [
+				'Authorization' => 'Bearer ' . $token,
+				'Content-Type'  => $content_type,
+			],
+			'body'    => $content,
+			'timeout' => 60,
+		] );
+
+		if ( is_wp_error( $response ) ) {
+			throw new Exception( 'Error subiendo archivo a SharePoint: ' . $response->get_error_message() );
+		}
+
+		$data = json_decode( wp_remote_retrieve_body( $response ), true );
+		if ( empty( $data['id'] ) ) {
+			$err = isset( $data['error']['message'] ) ? $data['error']['message'] : wp_remote_retrieve_body( $response );

+jQuery(document).ready(function($) {
+	if ($('#ipao-signature-pad').length === 0) return;
+
+	var canvas = document.getElementById('ipao-signature-pad');
+	
+	// Adjust canvas resolution for high-DPI displays
+	function resizeCanvas() {
+		var ratio =  Math.max(window.devicePixelRatio || 1, 1);
+		canvas.width = canvas.offsetWidth * ratio;
+		canvas.height = canvas.offsetHeight * ratio;
+		canvas.getContext("2d").scale(ratio, ratio);
+	}
+	window.addEventListener("resize", resizeCanvas);
+	resizeCanvas();
+
+	var signaturePad = new SignaturePad(canvas, {
+		backgroundColor: 'rgb(255, 255, 255)'
+	});
+
+	$('#ipao-clear-signature').on('click', function() {
+		signaturePad.clear();
+	});
+
+	$('#ipao-inscripcion-form').on('submit', function(e) {
+		e.preventDefault();
+
+		if (signaturePad.isEmpty()) {
+			alert("Por favor, proporciona tu firma.");
+			return;
+		}
+
+		var firmaDataUrl = signaturePad.toDataURL("image/png");
+		$('#ipao_firma_data').val(firmaDataUrl);
+
+		var $form = $(this);
+		var $submitBtn = $('#ipao-submit-btn');
+		var $spinner = $('.ipao-spinner');
+		var $messages = $('#ipao-form-messages');
+
+		$submitBtn.prop('disabled', true);
+		$spinner.show();
+		$messages.hide().removeClass('success error');
+
+		var formData = $form.serialize() + '&action=ipao_submit_form&nonce=' + ipao_ajax.nonce;
+
+		$.ajax({
+			type: 'POST',
+			url: ipao_ajax.url,
+			data: formData,
+			success: function(response) {
+				$submitBtn.prop('disabled', false);
+				$spinner.hide();
+
+				if (response.success) {
+					$messages.addClass('success').text(response.data.message);
+					$form[0].reset();
+					signaturePad.clear();
+				} else {
+					$messages.addClass('error').text(response.data.message || 'Error desconocido');
+				}
+			},
+			error: function() {
+				$submitBtn.prop('disabled', false);
+				$spinner.hide();
+				$messages.addClass('error').text('Error de conexión con el servidor.');
+			}
+		});
+	});
+});
